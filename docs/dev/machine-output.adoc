== application describe

[source,json]
----
{
  "kind": "Application",
  "apiVersion": "odo.openshift.io/v1alpha1",
  "metadata": {
    "name": "app",
    "namespace": "myproject",
    "creationTimestamp": null
  },
  "spec": {},
  "status": {
    "active": false
  }
}
----
